function twoSum(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left < right) {
        let sum = arr[left] + arr[right];
        if (sum === target) return [left, right];
        else if (sum < target) left++;
        else right--;
    }
    return [-1, -1];
}

// Example
let arr = [1, 2, 3, 4, 6];
console.log(twoSum(arr, 6)); // [1, 3]
